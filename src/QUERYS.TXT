select r.Rollo, w.Wrapper ,ws.DescripcionEstado, FechaRegistro, EndDate, dt.TiempoMin
from Wrapper_DownTimes dt
inner join Wrapper_Status ws on dt.IdWrapperStatus = ws.IdWrapper_Status
inner join Rollos r on r.idRollo = dt.idRollo
inner join Catalogo_Wrapper w on w.IdCatalogoWrapper = dt.IdWrapper
where r.idRollo in (23630, 23598)
order by r.idRollo asc */
